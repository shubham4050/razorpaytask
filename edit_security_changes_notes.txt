@csrf_exempt
def edit_card(request, pk):

        profile.website = request.POST.get("websiteurl")

        profile.address = request.POST.get("perm_address")

        profile.company = request.POST.get("companyname")
        profile.company_email = request.POST.get("companyemail")
        profile.company_phone1 = request.POST.get("companycontactno")
        profile.company_phone2 = request.POST.get("companycontactno2")
        profile.intro = request.POST.get("introtext")
        profile.company_tagline1 = request.POST.get("companytagline1")
        profile.company_tagline2 = request.POST.get("companytagline2")
        print("TAGLINE 1:", request.POST.get("companytagline1"))
        print("TAGLINE 2:", request.POST.get("companytagline2"))
        profile.personal_insta_url=request.POST.get('instagramurl')
        profile.personal_facebook_url=request.POST.get('facebookurl')
        profile.personal_linked_url=request.POST.get('linkdinurl')
        profile.personal_twitter_url=request.POST.get('twitterurl')

        profile.social_insta_url=request.POST.get('s_instagramurl')
        profile.social_facebook_url=request.POST.get('s_facebookurl')
        profile.social_linked_url=request.POST.get('s_linkdinurl')
        profile.social_twitter_url=request.POST.get('s_twitterurl')
        profile.social_pinterest_url=pinterest
        profile.social_youtube_url=youtube
        if company_logo:
            profile.company_logo=company_logo

        if request.FILES.get("banner_img"):
            profile.banner = request.FILES.get("banner_img")

        if request.FILES.get("photo"):
            profile.avatar = request.FILES.get("photo")

        profile.save()

        # ================= EDIT SERVICES (UPDATE + CREATE) =================

        titles = request.POST.getlist("servicetitle[]")
        taglines = request.POST.getlist("servicetagline[]")
        images = request.FILES.getlist("serviceimg[]")

        existing_services = list(Service.objects.filter(profile=profile))

        image_index = 0  # for uploaded images only

        for i in range(len(titles)):
            title = titles[i].strip()
            tagline = taglines[i].strip() if i < len(taglines) else ""

            if not title:
                continue

            # UPDATE EXISTING SERVICE
            if i < len(existing_services):
                service = existing_services[i]
                service.title = title
                service.description = tagline

                if image_index < len(images):
                    service.image = images[image_index]
                    image_index += 1

                service.save()

            # CREATE NEW SERVICE (from + button)
            else:
                image = None
                if image_index < len(images):
                    image = images[image_index]
                    image_index += 1

                Service.objects.create(
                    profile=profile,
                    title=title,
                    description=tagline,
                    image=image
                )

        # ================================
        # UPDATE BRANCHES
        # ================================

        Branch.objects.filter(profile=profile).delete()
        branch_names = request.POST.getlist("branch_name[]")
        for name in branch_names:
            if name.strip():
                Branch.objects.create(profile=profile, name=name)

        return redirect(f'/card/{profile.public_username}/')

    # ================================
    # GET REQUEST (LOAD EDIT PAGE)
    # ================================
    services = profile.services.all()
    branches = profile.branches.all()

    return render(request, "nk_editcard.html", {
        "card": profile,
        "profile": profile,
        "services": services,
        "branches": branches,
    })

def get_razorpay_client():
    if not settings.API_ID or not settings.API_SECRET:
        return None
    return razorpay.Client(auth=(settings.API_ID, settings.API_SECRET))


def create_order(request):
    razorpay_client = get_razorpay_client()
    if razorpay_client is None:
        return JsonResponse(
            {"status": "failed", "message": "Payment gateway is not configured."},
            status=503,
        )

    amount = settings.REGISTRATION_FEE * 100
    order = razorpay_client.order.create({
        "amount": amount,
        "currency": "INR",
        "payment_capture": 1
    })
    print('amount',amount)
    print('order',order)
    return JsonResponse({
        "order_id": order["id"],
        "amount": amount,
        "key": settings.API_ID
    })


@csrf_exempt
def verify_payment(request):
    if request.method != "POST":
        return JsonResponse({"status": "invalid"})

    razorpay_client = get_razorpay_client()
    if razorpay_client is None:
        return JsonResponse(
            {"status": "failed", "message": "Payment gateway is not configured."},
            status=503,
        )

    try:
        razorpay_client.utility.verify_payment_signature({
            "razorpay_order_id": request.POST.get("razorpay_order_id"),
            "razorpay_payment_id": request.POST.get("razorpay_payment_id"),
            "razorpay_signature": request.POST.get("razorpay_signature"),
        })

        return JsonResponse({"status": "success"})

    except SignatureVerificationError:
        return JsonResponse({"status": "failed"})

HOW SESSION ACCESS WORKS
========================

New record:
After successful payment, create_card saves:

request.session["latest_profile_id"] = profile.id

Then edit_card allows edit only when:

request.session.get("latest_profile_id") == profile.id

Old record through OTP:
generate_otp saves:

request.session["otp_profile_id"] = profile.id

Then edit_card allows edit only when:

request.session.get("otp_profile_id") == profile.id
