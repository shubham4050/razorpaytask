class PromoCode(models.Model):
    DISCOUNT_FLAT = "flat"
    DISCOUNT_PERCENT = "percent"
    DISCOUNT_TYPE_CHOICES = [
        (DISCOUNT_FLAT, "Flat"),
        (DISCOUNT_PERCENT, "Percent"),
    ]

    code = models.CharField(max_length=50, unique=True)
    razorpay_offer_id = models.CharField(max_length=100, unique=True)
    discount_type = models.CharField(max_length=10, choices=DISCOUNT_TYPE_CHOICES)
    discount_value = models.DecimalField(max_digits=10, decimal_places=2)
    max_discount_amount = models.DecimalField(max_digits=10, decimal_places=2, blank=True, null=True)
    minimum_order_amount = models.DecimalField(max_digits=10, decimal_places=2, default=0)
    max_uses_total = models.PositiveIntegerField(blank=True, null=True)
    max_uses_per_user = models.PositiveIntegerField(default=1)
    valid_from = models.DateTimeField(blank=True, null=True)
    valid_until = models.DateTimeField(blank=True, null=True)
    is_active = models.BooleanField(default=True)
    created_at = models.DateTimeField(auto_now_add=True)

    class Meta:
        ordering = ["code"]

    def __str__(self):
        return self.code

    def is_available_now(self):
        now = timezone.now()
        if not self.is_active:
            return False, "Promo code is inactive."
        if self.valid_from and now < self.valid_from:
            return False, "Promo code is not active yet."
        if self.valid_until and now > self.valid_until:
            return False, "Promo code has expired."
        if self.max_uses_total and self.usages.count() >= self.max_uses_total:
            return False, "Promo code usage limit is over."
        return True, ""


class PromoCodeUsage(models.Model):
    promo_code = models.ForeignKey(PromoCode, related_name="usages", on_delete=models.PROTECT)
    profile = models.ForeignKey(UserProfile, related_name="promo_usages", on_delete=models.CASCADE)
    razorpay_offer_id = models.CharField(max_length=100)
    razorpay_order_id = models.CharField(max_length=100)
    razorpay_payment_id = models.CharField(max_length=100, unique=True)
    amount = models.DecimalField(max_digits=10, decimal_places=2)
    used_at = models.DateTimeField(auto_now_add=True)

    class Meta:
        ordering = ["-used_at"]
        constraints = [
            models.UniqueConstraint(
                fields=["promo_code", "profile"],
                name="unique_promo_code_per_profile",
            )
        ]

    def __str__(self):
        return f"{self.profile} - {self.promo_code}"

def clean_int_id(value):
    value = str(value or "").strip()
    return value if value.isdigit() else None


def find_profile_from_request(request, profile_id=None):
    profile_id = clean_int_id(profile_id)
    if profile_id:
        profile = UserProfile.objects.filter(id=profile_id).first()
        if profile:
            return profile

    for session_key in ("current_profile_id", "latest_profile_id"):
        session_profile_id = clean_int_id(request.session.get(session_key))
        if session_profile_id:
            profile = UserProfile.objects.filter(id=session_profile_id).first()
            if profile:
                return profile

    public_username = (request.POST.get("public_username") or request.GET.get("public_username") or "").strip()
    if public_username:
        profile = UserProfile.objects.filter(public_username=public_username).first()
        if profile:
            return profile

    phone = (request.POST.get("contct_num") or request.GET.get("contct_num") or "").strip()
    if phone:
        profile = UserProfile.objects.filter(phone=phone).order_by("-id").first()
        if profile:
            return profile

    return UserProfile.objects.filter(payment_status=False).order_by("-id").first()

def create_card(request):

    context = {}

    if request.method == "POST":
        step = request.POST.get("step")
        posted_profile_id = clean_int_id(request.POST.get("profile_id"))
        session_profile_id = clean_int_id(request.session.get("current_profile_id"))
        latest_profile_id = clean_int_id(request.session.get("latest_profile_id"))
        profile_id = posted_profile_id

        # Handle Previous button first
        if request.POST.get("btn-prev"):
            profile_id = posted_profile_id or session_profile_id or latest_profile_id
            profile = UserProfile.objects.filter(id=profile_id).first() if profile_id else None
            print('max(int(step) - 1, 1): ', max(int(step) - 1, 1))
            context.update({
                "profile": profile,
                "active_step": max(int(step) - 1, 1)
            })
            return render(request, "nk_appform.html", context)

        # --- Step 1 Saving Logic ---
        if step == "1":
            full_name = request.POST.get("name")
            public_username = request.POST.get("public_username")
            phone = request.POST.get("contct_num")

            profile = None
            # Check if profile already exists
            if profile_id:
                profile = UserProfile.objects.filter(id=profile_id).first()

            if not profile:  # Only create new if profile does not exist
                if UserProfile.objects.filter(public_username=public_username).exists():
                    messages.error(request, 'Public URL already exists!')
                    return render(request, "nk_appform.html", context)
                elif UserProfile.objects.filter(phone=phone).exists():
                    messages.error(request, 'Phone already exists!')
                    return render(request, "nk_appform.html", context)

                profile = UserProfile.objects.create(
                    full_name=full_name,
                    public_username=public_username,
                    phone=phone,
                    phone2=request.POST.get("contct_num2", "").strip(),
                    title=request.POST.get("designation", "").strip(),
                    email=request.POST.get("email_Id", "").strip(),
                    address=request.POST.get("address", "").strip(),
                    personal_insta_url=request.POST.get("personal_insta_url", "").strip(),
                    personal_facebook_url=request.POST.get("personal_fac_url", "").strip(),
                    personal_linked_url=request.POST.get("personal_linked_url", "").strip(),
                    personal_twitter_url=request.POST.get("personal_twitter_url", "").strip(),
                    avatar=request.FILES.get("photo"),
                    banner=request.FILES.get("banner"),
                )
            else:
                # Update existing profile
                profile.full_name = full_name
                profile.phone = phone
                profile.phone2 = request.POST.get("contct_num2", "").strip()
                profile.title = request.POST.get("designation", "").strip()
                profile.email = request.POST.get("email_Id", "").strip()
                profile.address = request.POST.get("address", "").strip()
                profile.personal_insta_url = request.POST.get("personal_insta_url", "").strip()
                profile.personal_facebook_url = request.POST.get("personal_fac_url", "").strip()
                profile.personal_linked_url = request.POST.get("personal_linked_url", "").strip()
                profile.personal_twitter_url = request.POST.get("personal_twitter_url", "").strip()
                if request.FILES.get("photo"):
                    profile.avatar = request.FILES.get("photo")
                if request.FILES.get("banner"):
                    profile.banner = request.FILES.get("banner")
                profile.save()

            # After Save & Proceed → move to Step 2
            context.update({
                "profile": profile,
                "active_step": 2
            })
            request.session["current_profile_id"] = profile.id
            request.session["latest_profile_id"] = profile.id
            messages.success(request, "Step 1 saved successfully!")
            return render(request, "nk_appform.html", context)

        # ---------------- STEP 2 ----------------
        elif step == "2":

            profile_id = posted_profile_id or session_profile_id or latest_profile_id
            youtube = request.POST.get('youtube')
            pinterest = request.POST.get('pinterest')
            company_logo = request.FILES.get('company_logo')

            if not profile_id:
                context["error"] = "Complete Step 1 first."
                context["active_step"] = 1
                return render(request, "nk_appform.html", context)

            profile = get_object_or_404(UserProfile, id=profile_id)

            # Update company info
            profile.company = request.POST.get("companyname", "")
            profile.website = request.POST.get("website", "")
            profile.company_tagline1 = request.POST.get("companytagline1", "")
            profile.company_tagline2 = request.POST.get("companytagline2", "")
            profile.company_email = request.POST.get("company_email", "")
            profile.company_phone1 = request.POST.get("company_phone1", "")
            profile.company_phone2 = request.POST.get("company_phone2", "")
            profile.intro = request.POST.get("intro", "")

            # ----------- FIX: Save social URLs -----------
            profile.social_insta_url = request.POST.get("social_insta_url", "").strip() or request.POST.get("instagram", "").strip()
            profile.social_facebook_url = request.POST.get("social_facebook_url", "").strip() or request.POST.get("facebook", "").strip()
            profile.social_linked_url = request.POST.get("social_linked_url", "").strip() or request.POST.get("linkdin", "").strip()
            profile.social_twitter_url = request.POST.get("social_twitter_url", "").strip() or request.POST.get("twitter", "").strip()
            profile.social_youtube_url  = youtube
            profile.social_pinterest_url = pinterest
            if company_logo:
                profile.company_logo=company_logo
            # ---------------------------------------------

            profile.save()
            request.session["current_profile_id"] = profile.id
            request.session["latest_profile_id"] = profile.id

            # Branches
            profile.branches.all().delete()
            
            
            
            
            for name in request.POST.getlist("branch_name[]"):
                if name.strip():
                    Branch.objects.create(profile=profile, name=name.strip())

            # Services
            titles = request.POST.getlist("servicetitle[]")
            taglines = request.POST.getlist("servicetagline[]")
            images = request.FILES.getlist("serviceimg[]")

            profile.services.all().delete()

            for t, d, img in zip_longest(titles, taglines, images, fillvalue=""):
                if t.strip():
                    Service.objects.create(
                        profile=profile,
                        title=t.strip(),
                        description=d.strip(),
                        image=img
                    )

            # Social links table
            profile.social_links.all().delete()

            socials = {
                "LinkedIn": request.POST.get("linkdin"),
                "Instagram": request.POST.get("instagram"),
                "Facebook": request.POST.get("facebook"),
                "Twitter": request.POST.get("twitter"),
            }

            for platform, url in socials.items():
                if url:
                    SocialLink.objects.create(
                        profile=profile,
                        platform=platform,
                        url=url
                    )

            context.update({
                "profile": profile,
                "active_step": 3,
                # "success": "Step 2 saved successfully!"
            })
            messages.success(request,'Step 2 saved successfully!')

            return render(request, "nk_appform.html", context)

        # ---------------- STEP 3 (PAYMENT) ----------------
        elif step == "3":

            profile_id = posted_profile_id or session_profile_id or latest_profile_id
            profile = find_profile_from_request(request, profile_id)

def verify_otp(request):
    if request.method != "POST":
        return JsonResponse({'status': False, 'message': 'Invalid request'}, status=400)

    otp = "".join([
        request.POST.get('otp1', ''),
        request.POST.get('otp2', ''),
        request.POST.get('otp3', ''),
        request.POST.get('otp4', ''),
        request.POST.get('otp5', ''),
        request.POST.get('otp6', ''),
    ])

    if len(otp) != 6:
        return JsonResponse({'status': False, 'message': 'Enter complete OTP'}, status=400)

    profile_id = request.session.get('otp_profile_id')

    if not profile_id:
        return JsonResponse({'status': False, 'message': 'Session expired'}, status=403)

    profile = UserProfile.objects.get(id=profile_id)

    if not profile.otp:
        return JsonResponse({'status': False, 'message': 'OTP not generated'}, status=400)

    if profile.otp != otp:
        return JsonResponse({'status': False, 'message': 'Invalid OTP'}, status=400)

    if timezone.now() > profile.otp_expiry:
        return JsonResponse({'status': False, 'message': 'OTP expired'}, status=400)
    
    
    profile.otp = None
    profile.otp_expiry = None
    profile.save()

    request.session['otp_verified'] = True
    print("OTP RECEIVED:", otp)
    print("OTP SAVED:", profile.otp)


    # return JsonResponse({
    #     'status': True,
    #     'redirect_url': f"/card/{profile.public_username}/"
    
    if profile.payment_status:
        redirect_url = f"/card/{profile.public_username}/"
    else:
        redirect_url = f"/create/?profile_id={profile.id}"

    return JsonResponse({
        'status': True,
        'redirect_url': redirect_url
})

def validate_promo_for_profile(code, profile, amount):
    if not code:
        return None, ""

    normalized_code = code.strip().upper()
    promo = PromoCode.objects.filter(code__iexact=normalized_code).first()

    if not promo:
        return None, "Invalid promo code."

    is_available, message = promo.is_available_now()
    if not is_available:
        return None, message

    if Decimal(amount) < promo.minimum_order_amount:
        return None, f"Minimum order amount for this promo is ₹{promo.minimum_order_amount}."

    used_count = PromoCodeUsage.objects.filter(
        promo_code=promo,
        profile=profile,
    ).count()
    if used_count >= promo.max_uses_per_user:
        return None, "You have already used this promo code."

    return promo, ""

def create_order(request):
    if not razorpay_client:
        return JsonResponse({"status": "failed", "message": "Razorpay package is not installed."}, status=500)

    amount_rupees = Decimal(str(settings.REGISTRATION_FEE))
    amount = int(amount_rupees * 100)
    profile_id = clean_int_id(
        request.GET.get("profile_id")
        or request.POST.get("profile_id")
        or request.session.get("current_profile_id")
        or request.session.get("latest_profile_id")
    )
    promo_code = (
        request.GET.get("promo_code")
        or request.POST.get("promo_code")
        or request.session.get("applied_promo_code")
    )

    if not profile_id:
        profile = find_profile_from_request(request)
    else:
        profile = find_profile_from_request(request, profile_id)

    if not profile:
        return JsonResponse({"status": "failed", "message": "Complete Step 1 first."}, status=400)

    profile_id = str(profile.id)

    order_payload = {
        "amount": amount,
        "currency": "INR",
        "payment_capture": 1,
        "receipt": f"nexokard-{profile.id}",
        "notes": {
            "profile_id": str(profile.id),
            "promo_code": promo_code or "",
        },
    }

    promo = None
    if promo_code:
        promo, promo_error = validate_promo_for_profile(promo_code, profile, amount_rupees)
        if promo_error:
            return JsonResponse({"status": "failed", "message": promo_error}, status=400)

        order_payload["offers"] = [promo.razorpay_offer_id]
        order_payload["force_offer"] = True

    try:
        print("Razorpay order payload:", order_payload)
        order = razorpay_client.order.create(order_payload)
    except Exception as e:
        print("Razorpay order create error:", e)
        return JsonResponse({
            "status": "failed",
            "message": "Unable to create Razorpay order. Please check internet/Razorpay credentials and try again.",
        }, status=502)
    request.session[f"order_profile_{order['id']}"] = profile.id
    request.session["current_profile_id"] = profile.id

    if promo and profile_id:
        request.session[f"promo_order_{order['id']}"] = {
            "promo_id": promo.id,
            "profile_id": int(profile_id),
            "amount": str(amount_rupees),
        }

    print('amount',amount)
    print('order',order)
    response = {
        "order_id": order["id"],
        "amount": amount,
        "currency": order.get("currency", "INR"),
        "profile_id": profile.id,
        "key": settings.API_ID
    }
    if promo:
        response.update({
            "promo_code": promo.code,
            "offer_id": promo.razorpay_offer_id,
        })
    return JsonResponse(response)

@csrf_exempt
def verify_payment(request):
    if request.method != "POST":
        return JsonResponse({"status": "invalid"})
    if not razorpay_client:
        return JsonResponse({"status": "failed", "message": "Razorpay package is not installed."}, status=500)

    try:
        razorpay_client.utility.verify_payment_signature({
            "razorpay_order_id": request.POST.get("razorpay_order_id"),
            "razorpay_payment_id": request.POST.get("razorpay_payment_id"),
            "razorpay_signature": request.POST.get("razorpay_signature"),
        })

        order_id = request.POST.get("razorpay_order_id")
        payment_id = request.POST.get("razorpay_payment_id")
        order_profile_key = f"order_profile_{order_id}"
        profile_id = clean_int_id(request.session.get(order_profile_key))
        if profile_id:
            request.session["current_profile_id"] = int(profile_id)
            request.session["latest_profile_id"] = int(profile_id)

        promo_session_key = f"promo_order_{order_id}"
        promo_data = request.session.get(promo_session_key)

        if promo_data:
            promo = PromoCode.objects.filter(id=promo_data.get("promo_id")).first()
            profile = UserProfile.objects.filter(id=promo_data.get("profile_id")).first()

            if promo and profile:
                PromoCodeUsage.objects.get_or_create(
                    promo_code=promo,
                    profile=profile,
                    defaults={
                        "razorpay_offer_id": promo.razorpay_offer_id,
                        "razorpay_order_id": order_id,
                        "razorpay_payment_id": payment_id,
                        "amount": Decimal(str(promo_data.get("amount", settings.REGISTRATION_FEE))),
                    },
                )

            del request.session[promo_session_key]

        if order_profile_key in request.session:
            del request.session[order_profile_key]

        return JsonResponse({"status": "success", "profile_id": profile_id})
        

        

    except SignatureVerificationError:
        return JsonResponse({"status": "failed"})

let paymentInProgress = false;
let rzpInstance = null;
let promoApplied = false;

function getPromoParams(){
    return new URLSearchParams({
        profile_id: document.getElementById("profile_id").value || "",
        public_username: document.querySelector("[name='public_username']")?.value || "",
        contct_num: document.querySelector("[name='contct_num']")?.value || "",
        promo_code: document.getElementById("promo_code").value || ""
    });
}

function parseJsonResponse(res){
    return res.text().then(text => {
        let data = {};

        try {
            data = text ? JSON.parse(text) : {};
        } catch (e) {
            throw new Error("Server returned an HTML error page. Please restart Django and check the terminal error.");
        }

        if(!res.ok){
            throw new Error(data.message || "Request failed.");
        }

        return data;
    });
}

function applyPromoCode(btn){
    const promoInput = document.getElementById("promo_code");

    if(!promoInput.value.trim()){
        showPromoMessage("Enter promo code.", false);
        return;
    }

    btn.disabled = true;
    btn.innerText = "Checking...";

    fetch("/validate-promo/?" + getPromoParams().toString())
    .then(parseJsonResponse)
    .then(data => {
        promoApplied = true;
        if(data.profile_id){
            document.getElementById("profile_id").value = data.profile_id;
        }
        promoInput.value = data.promo_code || promoInput.value;
        showPromoMessage(data.message || "Promo code applied.", true);
    })
    .catch(err => {
        promoApplied = false;
        showPromoMessage(err.message || "Invalid promo code.", false);
    })
    .finally(() => {
        btn.disabled = false;
        btn.innerText = "Apply";
    });
}

function startPayment(btn){

    if(paymentInProgress) return;

    paymentInProgress = true;
    btn.disabled = true;
    btn.innerText = "Processing...";

    const params = getPromoParams();

    fetch("/create-order/?" + params.toString())
    .then(parseJsonResponse)
    .then(data => {
        showPromoMessage(data.promo_code ? "Promo code applied." : "", true);

        const options = {
            key: data.key,
            amount: data.amount,
            currency: data.currency || "INR",
            order_id: data.order_id,

            name: "Pixelstat eSolution",
            description: "1 Year Access",
            prefill: {
                name: document.querySelector("[name='name']")?.value || "",
                email: document.querySelector("[name='email_Id']")?.value || "",
                contact: document.querySelector("[name='contct_num']")?.value || ""
            },

            handler: function(response){
                verifyPayment(response);
            },

            modal: {
                ondismiss: function(){
                    sendPaymentFailed();
                    resetPaymentButton(btn);
                    rzpInstance = null;
                }
            }
        };

        rzpInstance = new Razorpay(options);
        rzpInstance.open();

    })
    .catch(err=>{
        console.error(err);
        showPromoMessage(err.message || "Payment failed to start", false);
        resetPaymentButton(btn);
    });
}

function showPromoMessage(message, isSuccess){
    const promoMessage = document.getElementById("promoMessage");
    if(!promoMessage) return;

    if(!message){
        promoMessage.style.display = "none";
        promoMessage.innerText = "";
        return;
    }

    promoMessage.style.display = "block";
    promoMessage.innerText = message;
    promoMessage.style.color = isSuccess ? "green" : "red";
    promoMessage.style.textAlign = "center";
    promoMessage.style.fontSize = "24px";
    promoMessage.style.margin = "12px 0 28px";
}

function sendPaymentFailed(){

    fetch("/create/", {
        method: "POST",
        headers: {
            "Content-Type": "application/x-www-form-urlencoded"
        },
        body: new URLSearchParams({
            step: "3",
            profile_id: document.getElementById("profile_id").value,
            payment_failed: "true"
        })
    });
}

function verifyPayment(response){

    fetch("/verify-payment/", {
        method: "POST",
        headers: {
            "Content-Type": "application/x-www-form-urlencoded"
        },
        body:
            "razorpay_payment_id=" + response.razorpay_payment_id +
            "&razorpay_order_id=" + response.razorpay_order_id +
            "&razorpay_signature=" + response.razorpay_signature
    })
    .then(parseJsonResponse)
    .then(data => {

        if(data.status === "success"){

            if(rzpInstance) rzpInstance.close();

            let form = document.getElementById("mainForm");

            if(data.profile_id){
                document.getElementById("profile_id").value = data.profile_id;
            }

            let input = document.createElement("input");
            input.type = "hidden";
            input.name = "razorpay_payment_id";
            input.value = response.razorpay_payment_id;

            form.appendChild(input);

            document.getElementById("step").value = "3";
            form.submit();

        } else {
            alert("Payment verification failed");
        }
    });
}

function resetPaymentButton(btn){
    paymentInProgress = false;
    if(btn){
        btn.disabled = false;
        btn.innerText = "Pay Now";
    }
}
