class CustomerQuery(models.Model):
    QUERY_TYPE_CHOICES = [
        ("payment", "Payment"),
        ("card_update", "Card Update"),
        ("login_otp", "Login / OTP"),
        ("cancellation", "Cancellation / Refund"),
        ("other", "Other"),
    ]
    STATUS_CHOICES = [
        ("pending", "Pending"),
        ("in_process", "In Process"),
        ("resolved", "Resolved"),
        ("closed", "Closed"),
    ]

    query_id = models.CharField(max_length=20, unique=True, blank=True)
    name = models.CharField(max_length=100, blank=True, null=True)
    contact_no = models.CharField(max_length=15)
    query_type = models.CharField(max_length=30, choices=QUERY_TYPE_CHOICES, default="payment")
    message = models.TextField(blank=True, null=True)
    proof_of_evidence = models.FileField(upload_to="query_evidence/", blank=True, null=True)
    status = models.CharField(max_length=20, choices=STATUS_CHOICES, default="pending")
    admin_name = models.CharField(max_length=100, blank=True, null=True)
    admin_comment = models.TextField(blank=True, null=True)
    admin_proof = models.FileField(upload_to="query_resolution/", blank=True, null=True)
    resolved_date = models.DateTimeField(blank=True, null=True)
    query_date = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)

    class Meta:
        ordering = ["-query_date"]
        verbose_name = "Customer Query"
        verbose_name_plural = "Customer Queries"

    def __str__(self):
        return f"{self.query_id} - {self.contact_no}"

    def save(self, *args, **kwargs):
        if not self.query_id:
            last_query = CustomerQuery.objects.order_by("-id").first()
            next_id = 1
            if last_query and last_query.query_id:
                last_number = "".join(filter(str.isdigit, last_query.query_id))
                if last_number:
                    next_id = int(last_number) + 1
            self.query_id = f"QRY{next_id:05d}"
        super().save(*args, **kwargs)

def clean_int_id(value):
    value = str(value or "").strip()
    return value if value.isdigit() else None


def normalize_mobile_number(value):
    digits = "".join(ch for ch in str(value or "").strip() if ch.isdigit())
    if len(digits) == 12 and digits.startswith("91"):
        digits = digits[2:]
    return digits if len(digits) == 10 else ""


def is_profile_edit_authorized(request, profile):
    if request.user.is_authenticated and request.user.is_staff:
        return True

    verified_profile_id = clean_int_id(
        request.session.get("verified_edit_profile_id")
        or request.session.get("otp_profile_id")
    )

    return bool(
        request.session.get("otp_verified")
        and verified_profile_id
        and str(profile.id) == str(verified_profile_id)
    )


def apply_date_filters(queryset, request, date_field="created_at"):
    start_date = parse_date(request.GET.get("start_date") or "")
    end_date = parse_date(request.GET.get("end_date") or "")

    if start_date:
        queryset = queryset.filter(**{f"{date_field}__date__gte": start_date})
    if end_date:
        queryset = queryset.filter(**{f"{date_field}__date__lte": end_date})

    return queryset


def paginate_items(request, queryset, per_page=15):
    page = request.GET.get("page", 1)
    paginator = Paginator(queryset, per_page)
    return paginator.get_page(page)


def AdminManagerRole(user):
    return user.is_active and user.is_staff

6. cardapp/views.py - raise_query
==================================================================

Replace/merge your raise_query view with this.
Important details:
- It normalizes contact number to 10 digits.
- It does NOT auto-fill the form fields in template.
- If user is logged in, sidebar can still show My Card / Query Status.

def raise_query(request):
    profile = get_verified_profile_from_session(request)
    contexts = {
        "profile": profile,
        "can_edit_card": is_profile_edit_authorized(request, profile) if profile else False,
    }

    if request.method == "POST":
        name = request.POST.get("name", "").strip()
        contact_no = request.POST.get("contact_no", "").strip()
        normalized_contact_no = normalize_mobile_number(contact_no)
        query_type = request.POST.get("query_type", "").strip()
        message = request.POST.get("message", "").strip()
        proof_of_evidence = request.FILES.get("proof_of_evidence")

        if not normalized_contact_no:
            messages.error(request, "Valid 10 digit contact number is required.")
            return render(request, "nk_raise_query.html", contexts)

        query = CustomerQuery.objects.create(
            name=name,
            contact_no=normalized_contact_no,
            query_type=query_type or "payment",
            message=message,
            proof_of_evidence=proof_of_evidence,
        )
        messages.success(request, f"Your query has been raised successfully. Query ID: {query.query_id}")
        return redirect("raise_query")

    return render(request, "nk_raise_query.html", contexts)

def get_verified_profile_from_session(request):
    profile_id = clean_int_id(
        request.session.get("verified_edit_profile_id")
        or request.session.get("otp_profile_id")
        or request.session.get("current_profile_id")
        or request.session.get("latest_profile_id")
    )
    if not profile_id:
        return None
    return UserProfile.objects.filter(id=profile_id).first()


def query_status(request):
    profile = get_verified_profile_from_session(request)
    if not profile or not request.session.get("otp_verified"):
        messages.warning(request, "Please login with OTP to view your query status.")
        return redirect("index")

    contact_no = normalize_mobile_number(profile.phone)
    queries = CustomerQuery.objects.filter(contact_no=contact_no).order_by("-query_date")

    contexts = {
        "profile": profile,
        "queries": queries,
        "can_edit_card": is_profile_edit_authorized(request, profile),
    }
    return render(request, "nk_query_status.html", contexts)

@user_passes_test(AdminManagerRole, login_url="/manager/login/")
def manager_queries(request):
    queries = CustomerQuery.objects.all()
    queries = apply_date_filters(queries, request, "query_date")
    contact_no = (request.GET.get("contact_no") or "").strip()
    status = (request.GET.get("status") or "").strip()
    query_type = (request.GET.get("query_type") or "").strip()

    if contact_no:
        queries = queries.filter(contact_no__icontains=contact_no)
    if status:
        queries = queries.filter(status=status)
    if query_type:
        queries = queries.filter(query_type=query_type)

    query_params = request.GET.copy()
    query_params.pop("page", None)

    contexts = {
        "active_tab": "queries",
        "page_items": paginate_items(request, queries),
        "status_choices": CustomerQuery.STATUS_CHOICES,
        "query_type_choices": CustomerQuery.QUERY_TYPE_CHOICES,
        "query_string": query_params.urlencode(),
    }
    return render(request, "manager/queries.html", contexts)


Add manager_view_query.
Purpose:
- Admin opens one query.
- Admin changes status/comment/resolver/proof.
- resolved_date is set when status is resolved/closed.

@user_passes_test(AdminManagerRole, login_url="/manager/login/")
def manager_view_query(request, query_id):
    query = get_object_or_404(CustomerQuery, id=query_id)

    if request.method == "POST":
        status = request.POST.get("status")
        admin_name = request.POST.get("admin_name", "").strip()
        admin_comment = request.POST.get("admin_comment", "").strip()
        admin_proof = request.FILES.get("admin_proof")

        if status not in dict(CustomerQuery.STATUS_CHOICES):
            messages.error(request, "Select a valid query status.")
            return redirect("manager_view_query", query_id=query.id)

        query.status = status
        query.admin_name = admin_name
        query.admin_comment = admin_comment
        if admin_proof:
            query.admin_proof = admin_proof
        if status in ("resolved", "closed") and not query.resolved_date:
            query.resolved_date = timezone.now()
        elif status not in ("resolved", "closed"):
            query.resolved_date = None
        query.save()

        messages.success(request, "Query details updated successfully.")
        return redirect("manager_view_query", query_id=query.id)

    contexts = {
        "active_tab": "queries",
        "query": query,
        "status_choices": CustomerQuery.STATUS_CHOICES,
    }
    return render(request, "manager/view_query.html", contexts)
