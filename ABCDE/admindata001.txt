class CustomerQuery(models.Model):
    QUERY_TYPE_CHOICES = [
        ("payment", "Payment"),
        ("card_update", "Card Update"),
        ("login_otp", "Login / OTP"),
        ("cancellation", "Cancellation / Refund"),
        ("other", "Other"),
    ]
    STATUS_CHOICES = [
        ("pending", "Pending"),
        ("in_process", "In Process"),
        ("resolved", "Resolved"),
        ("closed", "Closed"),
    ]

    query_id = models.CharField(max_length=20, unique=True, blank=True)
    name = models.CharField(max_length=100, blank=True, null=True)
    contact_no = models.CharField(max_length=15)
    query_type = models.CharField(max_length=30, choices=QUERY_TYPE_CHOICES, default="payment")
    message = models.TextField(blank=True, null=True)
    proof_of_evidence = models.FileField(upload_to="query_evidence/", blank=True, null=True)
    status = models.CharField(max_length=20, choices=STATUS_CHOICES, default="pending")
    admin_name = models.CharField(max_length=100, blank=True, null=True)
    admin_comment = models.TextField(blank=True, null=True)
    admin_proof = models.FileField(upload_to="query_resolution/", blank=True, null=True)
    resolved_date = models.DateTimeField(blank=True, null=True)
    query_date = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)

    class Meta:
        ordering = ["-query_date"]
        verbose_name = "Customer Query"
        verbose_name_plural = "Customer Queries"

    def __str__(self):
        return f"{self.query_id} - {self.contact_no}"

    def save(self, *args, **kwargs):
        if not self.query_id:
            last_query = CustomerQuery.objects.order_by("-id").first()
            next_id = 1
            if last_query and last_query.query_id:
                last_number = "".join(filter(str.isdigit, last_query.query_id))
                if last_number:
                    next_id = int(last_number) + 1
            self.query_id = f"QRY{next_id:05d}"
        super().save(*args, **kwargs)


class PromoCode(models.Model):
    DISCOUNT_FLAT = "flat"
    DISCOUNT_PERCENT = "percent"
    DISCOUNT_TYPE_CHOICES = [
        (DISCOUNT_FLAT, "Flat"),
        (DISCOUNT_PERCENT, "Percent"),
    ]

    code = models.CharField(max_length=50, unique=True)
    razorpay_offer_id = models.CharField(max_length=100, unique=True)
    discount_type = models.CharField(max_length=10, choices=DISCOUNT_TYPE_CHOICES)
    discount_value = models.DecimalField(max_digits=10, decimal_places=2)
    max_discount_amount = models.DecimalField(max_digits=10, decimal_places=2, blank=True, null=True)
    minimum_order_amount = models.DecimalField(max_digits=10, decimal_places=2, default=0)
    max_uses_total = models.PositiveIntegerField(blank=True, null=True)
    max_uses_per_user = models.PositiveIntegerField(default=1)
    valid_from = models.DateTimeField(blank=True, null=True)
    valid_until = models.DateTimeField(blank=True, null=True)
    is_active = models.BooleanField(default=True)
    created_at = models.DateTimeField(auto_now_add=True)

    class Meta:
        ordering = ["code"]

    def __str__(self):
        return self.code

    def is_available_now(self):
        now = timezone.now()
        if not self.is_active:
            return False, "Promo code is inactive."
        if self.valid_from and now < self.valid_from:
            return False, "Promo code is not active yet."
        if self.valid_until and now > self.valid_until:
            return False, "Promo code has expired."
        if self.max_uses_total and self.usages.count() >= self.max_uses_total:
            return False, "Promo code usage limit is over."
        return True, ""


class PromoCodeUsage(models.Model):
    promo_code = models.ForeignKey(PromoCode, related_name="usages", on_delete=models.PROTECT)
    profile = models.ForeignKey(UserProfile, related_name="promo_usages", on_delete=models.CASCADE)
    razorpay_offer_id = models.CharField(max_length=100)
    razorpay_order_id = models.CharField(max_length=100)
    razorpay_payment_id = models.CharField(max_length=100, unique=True)
    amount = models.DecimalField(max_digits=10, decimal_places=2)
    used_at = models.DateTimeField(auto_now_add=True)

    class Meta:
        ordering = ["-used_at"]
        constraints = [
            models.UniqueConstraint(
                fields=["promo_code", "profile"],
                name="unique_promo_code_per_profile",
            )
        ]

    def __str__(self):
        return f"{self.profile} - {self.promo_code}"

def AdminManagerRole(user):
    return user.is_active and user.is_staff


def manager_login(request):
    if request.user.is_authenticated and request.user.is_staff:
        return redirect("manager_dashboard")

    next_url = request.GET.get("next") or request.POST.get("next") or "manager_dashboard"

    if request.method == "POST":
        username = request.POST.get("username", "").strip()
        password = request.POST.get("password", "")
        user = authenticate(request, username=username, password=password)

        if user and user.is_active and user.is_staff:
            login(request, user)
            return redirect(next_url)

        messages.error(request, "Invalid manager login details.")

    return render(request, "manager/login.html", {"next": next_url})


def manager_logout(request):
    logout(request)
    return redirect("manager_login")


def apply_date_filters(queryset, request, date_field="created_at"):
    start_date = parse_date(request.GET.get("start_date") or "")
    end_date = parse_date(request.GET.get("end_date") or "")

    if start_date:
        queryset = queryset.filter(**{f"{date_field}__date__gte": start_date})
    if end_date:
        queryset = queryset.filter(**{f"{date_field}__date__lte": end_date})

    return queryset


def paginate_items(request, queryset, per_page=15):
    page = request.GET.get("page", 1)
    paginator = Paginator(queryset, per_page)
    return paginator.get_page(page)


def filter_profiles(request, queryset):
    queryset = apply_date_filters(queryset, request, "created_at")
    contact_no = (request.GET.get("contact_no") or "").strip()
    payment_id = (request.GET.get("payment_id") or "").strip()
    public_username = (request.GET.get("public_username") or "").strip()

    if contact_no:
        queryset = queryset.filter(phone__icontains=contact_no)
    if payment_id:
        queryset = queryset.filter(razorpay_payment_id__icontains=payment_id)
    if public_username:
        queryset = queryset.filter(public_username__icontains=public_username)

    return queryset.order_by("-created_at")

@user_passes_test(AdminManagerRole, login_url="/manager/login/")
def manager_view_query(request, query_id):
    query = get_object_or_404(CustomerQuery, id=query_id)

    if request.method == "POST":
        status = request.POST.get("status")
        admin_name = request.POST.get("admin_name", "").strip()
        admin_comment = request.POST.get("admin_comment", "").strip()
        admin_proof = request.FILES.get("admin_proof")

        if status not in dict(CustomerQuery.STATUS_CHOICES):
            messages.error(request, "Select a valid query status.")
            return redirect("manager_view_query", query_id=query.id)

        query.status = status
        query.admin_name = admin_name
        query.admin_comment = admin_comment
        if admin_proof:
            query.admin_proof = admin_proof
        if status in ("resolved", "closed") and not query.resolved_date:
            query.resolved_date = timezone.now()
        elif status not in ("resolved", "closed"):
            query.resolved_date = None
        query.save()

        messages.success(request, "Query details updated successfully.")
        return redirect("manager_view_query", query_id=query.id)

    contexts = {
        "active_tab": "queries",
        "query": query,
        "status_choices": CustomerQuery.STATUS_CHOICES,
    }
    return render(request, "manager/view_query.html", contexts)


def parse_manager_datetime(value):
    if not value:
        return None
    parsed = parse_datetime(value)
    if parsed and timezone.is_naive(parsed):
        parsed = timezone.make_aware(parsed)
    return parsed

def manager_promo_codes(request, promo_id=None):
    promo = PromoCode.objects.filter(id=promo_id).first() if promo_id else None

    if request.method == "POST":
        code = (request.POST.get("code") or "").strip().upper()
        razorpay_offer_id = (request.POST.get("razorpay_offer_id") or "").strip()
        discount_type = request.POST.get("discount_type") or PromoCode.DISCOUNT_FLAT
        discount_value = Decimal(request.POST.get("discount_value") or "0")
        max_discount_amount = request.POST.get("max_discount_amount") or None
        minimum_order_amount = Decimal(request.POST.get("minimum_order_amount") or "0")
        max_uses_total = request.POST.get("max_uses_total") or None
        max_uses_per_user = request.POST.get("max_uses_per_user") or 1
        is_active = request.POST.get("is_active") == "on"

        if not promo:
            promo = PromoCode()

        promo.code = code
        promo.razorpay_offer_id = razorpay_offer_id
        promo.discount_type = discount_type
        promo.discount_value = discount_value
        promo.max_discount_amount = Decimal(max_discount_amount) if max_discount_amount else None
        promo.minimum_order_amount = minimum_order_amount
        promo.max_uses_total = int(max_uses_total) if max_uses_total else None
        promo.max_uses_per_user = int(max_uses_per_user)
        promo.valid_from = parse_manager_datetime(request.POST.get("valid_from"))
        promo.valid_until = parse_manager_datetime(request.POST.get("valid_until"))
        promo.is_active = is_active
        promo.save()
        messages.success(request, "Promo code saved successfully.")
        return redirect("manager_promo_codes")

    contexts = {
        "active_tab": "promo_codes",
        "promo": promo,
        "promo_codes": PromoCode.objects.all(),
        "discount_type_choices": PromoCode.DISCOUNT_TYPE_CHOICES,
    }
    return render(request, "manager/promo_codes.html", contexts)
