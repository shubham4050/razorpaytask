Purpose:
This file contains only today's Add to Contact / .vcf changes for the public
business card page:

    path("bc/<str:public_username>/", views.business_card, name="business_card")

When the user clicks the corner user-plus button, a custom Add to Contact tab opens
with only one option: Add to contacts. That option downloads/opens a .vcf contact
file. On mobile, this opens the phone contact-save screen with data already filled.

Important:
A normal website cannot silently save contacts directly into Android/iPhone
contacts. The correct web flow is:

    Click Add to Contact -> open/download .vcf -> mobile shows Save Contact screen -> user saves

```
# VS Changes - vCard helpers for Add to Contact button on public business card.
def escape_vcard_value(value):
    value = str(value or "").strip()
    return (
        value.replace("\\", "\\\\")
        .replace(";", "\\;")
        .replace(",", "\\,")
        .replace("\r\n", "\\n")
        .replace("\n", "\\n")
        .replace("\r", "\\n")
    )


# VS Changes - download contact details as .vcf so mobile opens pre-filled Save Contact screen.
def download_business_card_vcard(request, public_username):
    profile = get_object_or_404(UserProfile, public_username=public_username)
    full_name = escape_vcard_value(profile.full_name)
    company = escape_vcard_value(profile.company)
    title = escape_vcard_value(profile.title)
    address = escape_vcard_value(profile.address)
    intro = escape_vcard_value(profile.intro)
    website = escape_vcard_value(profile.website)

    vcard_lines = [
        "BEGIN:VCARD",
        "VERSION:3.0",
        f"FN:{full_name}",
    ]

    if company:
        vcard_lines.append(f"ORG:{company}")
    if title:
        vcard_lines.append(f"TITLE:{title}")
    if profile.phone:
        vcard_lines.append(f"TEL;TYPE=CELL:{escape_vcard_value(profile.phone)}")
    if profile.phone2:
        vcard_lines.append(f"TEL;TYPE=CELL:{escape_vcard_value(profile.phone2)}")
    if profile.company_phone1:
        vcard_lines.append(f"TEL;TYPE=WORK:{escape_vcard_value(profile.company_phone1)}")
    if profile.company_phone2:
        vcard_lines.append(f"TEL;TYPE=WORK:{escape_vcard_value(profile.company_phone2)}")
    if profile.email:
        vcard_lines.append(f"EMAIL;TYPE=HOME:{escape_vcard_value(profile.email)}")
    if profile.company_email:
        vcard_lines.append(f"EMAIL;TYPE=WORK:{escape_vcard_value(profile.company_email)}")
    if website:
        vcard_lines.append(f"URL:{website}")
    if address:
        vcard_lines.append(f"ADR;TYPE=WORK:;;{address};;;;")
    if intro:
        vcard_lines.append(f"NOTE:{intro}")

    vcard_lines.append("END:VCARD")
    vcard_data = "\r\n".join(vcard_lines) + "\r\n"
    filename = f"{profile.public_username or 'business_card'}.vcf"

    response = HttpResponse(vcard_data, content_type="text/vcard; charset=utf-8")
    response["Content-Disposition"] = f'attachment; filename="{filename}"'
    return response
```

Fields used from UserProfile:

    full_name
    company
    title
    phone
    phone2
    company_phone1
    company_phone2
    email
    company_email
    website
    address
    intro
    public_username

If your other build has different field names, update only those profile.field
references inside download_business_card_vcard.
